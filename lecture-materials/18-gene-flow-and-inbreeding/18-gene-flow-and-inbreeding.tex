             & \textit{C.\ elegans} & \textit{E.\ coli} \\
            \hline

            \clickeranswer{$\approx$33 hours} \\

            \textcolor{red}{5)} & \multicolumn{2}{l}{It depends on the number of genes in the genome} \\ 
        \end{tabular}
    \end{table}

\section[How does gene flow cause changes in allele frequencies?]{How gene flow changes allele frequencies}

\begin{frame}[t]
    \frametitle{Gene flow: The one-island model}
    \begin{adjustwidth}{-1.5em}{-1.5em}
        \vspace{-4mm}
        Model: One-way movement of individuals (and their alleles) from a large
        population on a continent to a small population on an island.

        \uncover<2->{
        \vspace{2mm}
        Suppose that allele $A_1$ is at frequency 1.0 on an island with 400
        individuals.
        }

        \uncover<3->{
        \vspace{1mm}
        Suppose that allele $A_2$ is at frequency 1.0 on the continent.
        }

        \uncover<4->{
        \vspace{1mm}
        If 100 individuals from the continent migrate to the island, the
        frequencies of alleles in the island populations are now (the species
        is diploid):
        }

        \nbox{$fr(A_2) = \frac{100 \times 2}{1000} = 0.2$ \\
              $fr(A_1) = \frac{400 \times 2}{1000} = 0.8$} 

    
        \uncover<5->{
        \vspace{2cm}
        Punchline: Evolution has occurred because of gene flow!
        }

    \end{adjustwidth}
\end{frame}
